# 定时函数<body中>
* 每隔1min触发

```html
<script>
    function get_withdraw_msg() {
        $.get('/crm2/pay/withdraw/operate_msg', function (res) {
            if (res.code == -1) {

            } else if (res.code) {
                showMsg(res.msg, 'danger');
            } else {
                showMsg(res.msg, 'success');
            }
        })
    }

    setInterval(get_withdraw_msg, 6000);
</script>
```

```html
function showMsg( text, type) {
    type = type || 'warning';
    text = text || 'no text';
    $('#msg_alert_text').text(text);
    $('#msg_alert').removeAttr('class').attr('class', 'alert alert-'+type).show();
    clearTimeout(MSG_TIMEOUT);
    MSG_TIMEOUT = setTimeout( function(){
        $('#msg_alert').hide();
    }, 3000 );
}
```
